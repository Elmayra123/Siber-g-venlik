    <div class="section">
        <h2>1. Parola vs. Şifre (Cipher) Ayrımı</h2>
        <p>Giriş yaparken kullandığınız şey <b>Parola</b>'dır. <b>Şifre (Cipher)</b> ise veriyi gizleme algoritmasıdır.</p>
        <input type="text" id="cipherInput" placeholder="Şifrelenecek düz metni yazın..." oninput="simulateCipher()">
        <div class="result">🕵️‍♂️ Şifrelenmiş Metin (Cipher Text): <span id="cipherResult">-</span></div>

    <div class="section">
        <h2>2. Kırılmaz Parola Üretim Hattı</h2>
        <p>Akılda kalıcı bir cümle girin, sistem onu mutasyona uğratsın.</p>
        <input type="text" id="sentenceInput" placeholder="Örn: Bayılırım biber dolmasına Perşembeleri kahvaltıda">
        <button onclick="generateSecurePassword()">Parola Üret</button>
        <div class="result">🛠️ Üretilen Güçlü Parola: <span id="generatedPassword">-</span></div>
    </div>

        <h2>3. Sistemin Arka Yüzü: Tuzlama & Hash</h2>

<script>
    // 1. Bölüm: Basit bir Sezar Şifreleme simülasyonu (3 harf kaydırma)
    function simulateCipher() {
        let text = document.getElementById('cipherInput').value;
        let ciphered = text.split('').map(char => {
            let code = char.charCodeAt(0);
            return String.fromCharCode(code + 3); // Basit mutasyon
        }).join('');
        document.getElementById('cipherResult').innerText = text ? ciphered : "-";
    }

    // 2. ve 3. Bölüm: Cümleden Parola ve Hash üretimi
    function generateSecurePassword() {
        let sentence = document.getElementById('sentenceInput').value;
        if(!sentence) return alert("Lütfen bir cümle girin!");

        // Kelimelerin ilk ve son harflerini al
        let words = sentence.split(' ');
        let basePassword = words.map(word => {
            if(word.length > 1) return word[0] + word[word.length - 1];
            return word[0] || '';
        }).join('');

        // Güçlendirme (Mutasyon) B->8, b->6, i->!, a->@
        let securePassword = basePassword
            .replace(/B/g, '8').replace(/b/g, '6')
            .replace(/i/g, '!').replace(/ı/g, '!')
            .replace(/a/g, '@').replace(/o/g, '0');

        document.getElementById('generatedPassword').innerText = securePassword;

        // Arka yüz simülasyonu (Hash üretimi)
        let salt = "salt_" + Math.random().toString(36).substring(2, 7);
        document.getElementById('saltResult').innerText = salt;
        
        // Basit bir hash görüntüsü simüle edelim (Gerçek SHA için kütüphane gerekir)
        let simulatedHash = btoa(securePassword + salt).substring(0, 16) + "...(SHA-256)";
        document.getElementById('hashResult').innerText = simulatedHash;
    }
</script>
